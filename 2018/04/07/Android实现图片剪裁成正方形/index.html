Android实现图片剪裁成正方形

作者:PJR    发布于: 2018-04-07

虽然上学期学过安卓，但是真正实际运用中还是会遇到很多问题。比如最近帮师兄做毕设的时候就问题连连，故将一些问题记录于此，以便后来复习参考。
在实现朋友圈的时候，朋友圈的图片显示都是正方形的，所以需要进行裁剪，一下便是裁剪的详细过程。

import android.graphics.Bitmap;

public class ImageUtils {

    private Bitmap bitmap;
    private int edgeLength;

    public ImageUtils(Bitmap bitmap,int edgeLength) {
        this.bitmap = bitmap;
        this.edgeLength = edgeLength;
    }

    public Bitmap centerSquareScaleBitmap()
    {
        if(null == bitmap || edgeLength <= 0)
        {
            return  null;
        }

        Bitmap result = bitmap;
        int widthOrg = bitmap.getWidth();
        int heightOrg = bitmap.getHeight();

        if(widthOrg > edgeLength && heightOrg > edgeLength)
        {
            //压缩到一个最小长度是edgeLength的bitmap
            int longerEdge = (int)(edgeLength * Math.max(widthOrg, heightOrg) / Math.min(widthOrg, heightOrg));
            int scaledWidth = widthOrg > heightOrg ? longerEdge : edgeLength;
            int scaledHeight = widthOrg > heightOrg ? edgeLength : longerEdge;
            Bitmap scaledBitmap;

            try{
                scaledBitmap = Bitmap.createScaledBitmap(bitmap, scaledWidth, scaledHeight, true);
            }
            catch(Exception e){
                return null;
            }

            //从图中截取正中间的正方形部分。
            int xTopLeft = (scaledWidth - edgeLength) / 2;
            int yTopLeft = (scaledHeight - edgeLength) / 2;

            try{
                result = Bitmap.createBitmap(scaledBitmap, xTopLeft, yTopLeft, edgeLength, edgeLength);
                scaledBitmap.recycle();
            }
            catch(Exception e){
                return null;
            }
        }

        return result;
    }
}
